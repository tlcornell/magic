Startup:
	irparam sys.wall 1000
	closest = sys.wall
	jmpoffset = add JumpTableBase closest
	ircall sys.wall ArrivedAtWall
	irparam sys.wall 30
	irstart
	jump jmpoffset
JumpTableBase:
	log "unreachable"
	jump StartMovingNorth
	jump StartMovingWest
	jump StartMovingSouth
	jump StartMovingEast

StartMovingNorth:
	sys.velocity_dy = -5
	jump Traveling
StartMovingWest:
	sys.velocity_dx = -5
	jump Traveling
StartMovingSouth:
	sys.velocity_dy = 5
	jump Traveling
StartMovingEast:
	sys.velocity_dx = 5
	jump Traveling

Traveling:
	jump Traveling

ArrivedAtWall:
	sys.velocity_dx = 0
	sys.velocity_dy = 0
	ircall sys.agents Shoot
	irparam sys.agents 10000
	ircall sys.wall -1
	jmpoffset = add WallTurretBase sys.wall
	jump jmpoffset
WallTurretBase:
	log "better be unreachable"
	ircontinue OnNorthWall
	ircontinue OnWestWall
	ircontinue OnSouthWall
