Startup:
#	debug

	irparam sys.wall 1000
	closest = sys.wall
	jmpoffset = add JumpTableBase closest
	ircall sys.wall ArrivedAtWall
	irparam sys.wall 30
	irstart
	jump jmpoffset
JumpTableBase:
	log "unreachable"
	jump StartMovingNorth
	jump StartMovingWest
	jump StartMovingSouth
	jump StartMovingEast

StartMovingNorth:
	sys.velocity_dy = -5
	jump Traveling
StartMovingWest:
	sys.velocity_dx = -5
	jump Traveling
StartMovingSouth:
	sys.velocity_dy = 5
	jump Traveling
StartMovingEast:
	sys.velocity_dx = 5
	jump Traveling

Traveling:
	jump Traveling

ArrivedAtWall:
	sys.velocity_dx = 0
	sys.velocity_dy = 0
	ircall sys.agents Shoot
	irparam sys.agents 10000
	ircall sys.wall RegainPosition
	jmpoffset = add WallTurretBase sys.wall
	jump jmpoffset
WallTurretBase:
	log "better be unreachable"
	ircontinue OnNorthWall
	ircontinue OnWestWall
	ircontinue OnSouthWall
	ircontinue OnEastWall

Shoot:
	sys.aim = add sys.aim sys.agents.angle
	A = gte sys.energy 50
	ifz A SkipIt
	sys.fire = 50
SkipIt:
	ircontinue

RegainPosition:
	irparam sys.wall 10000
	jmpoffset = add RegainBase sys.wall
	irparam sys.wall 30
	jump jmpoffset
RegainBase:
	log "noop"
	jump GetOffNorth
	jump GetOffWest
	jump GetOffSouth
	jump GetOffEast
GetOffNorth:
	log "GetOffNorth"
	sys.velocity_dy = 2
	sync
	A = lte sys.y 50
	log sys.y 50
	if A GetOffNorth
	sys.velocity_dy = 0
	ircontinue
GetOffWest:
	log "GetOffWest"
	sys.velocity_dx = 2
	sync
	A = lte sys.x 50
	log sys.x 50
	if A GetOffWest
	sys.velocity_dx = 0
	ircontinue
GetOffSouth:
	log "GetOffSouth"
	sys.velocity_dy = -2
	sync
	A = gte sys.y 590
	log sys.y 590
	if A GetOffSouth
	sys.velocity_dy = 0
	ircontinue
GetOffEast:
	log "GetOffEast"
	sys.velocity_dx = -2
	sync
	A = gte sys.x 750
	log sys.x 750
	if A GetOffEast
	sys.velocity_dx = 0
	ircontinue
