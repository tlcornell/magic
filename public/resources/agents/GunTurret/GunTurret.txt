# GunTurret
#
# For debugging the vision system
#

Main:
	sys.mv.velocity = tuple 0 0
	if sys.agents DoFire KeepLooking

DoFire:
	lowEnergy = sys.energy < 20
	if lowEnergy DontFire
	sys.fire = 30
	jump Main
DontFire:
	sync	# let us recharge a bit
	jump Main

KeepLooking:
	sys.aim = sys.aim + 13
	jump Main
