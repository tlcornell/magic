# Navigator

#	debug
	
##########################################################3
# Find the closest wall

	irparam sys.wall 1000
	closest = sys.wall
	jmpoffset = add StartTableBase closest
	irparam sys.wall 20
	ircall sys.wall HugWall
	irstart
	jump jmpoffset
StartTableBase:
	log "unreachable"
	jump GoNorth
	jump GoWest
	jump GoSouth
	jump GoEast

GoNorth:
	sys.velocity_dy = -3 
	jump SeekingWall
GoWest:
	sys.velocity_dx = -3 
	jump SeekingWall
GoSouth:
	sys.velocity_dy = 3 
	jump SeekingWall
GoEast:
	sys.velocity_dx = 3 
	jump SeekingWall


SeekingWall:
	call CheckRange
	jump SeekingWall
	

HugWall:
	dx = sys.velocity_dx
	sys.velocity_dx = sys.velocity_dy
	sys.velocity_dy = dx
	ircontinue HugLoop	# pop stack frame, re-enable interrupts
HugLoop:
	ircall sys.wall ReverseDirection
	call CheckRange
	jump HugLoop

ReverseDirection:
	# One of dx or dy is zero, so this is okay
	sys.velocity_dx = mul -1 sys.velocity_dx
	sys.velocity_dy = mul -1 sys.velocity_dy
	ircontinue

CheckRange:
	A = gt sys.agents.data.dist 0 
	ifnz A DoFire DoRotate
DoFire:
	A = gte sys.energy 10
	ifz A FireReturn
	sys.fire = 10 
FireReturn:
	return
DoRotate:
	sys.aim = add sys.aim 7 
	return
